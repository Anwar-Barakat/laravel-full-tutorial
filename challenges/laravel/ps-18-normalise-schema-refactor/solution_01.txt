// ============================================================
// SOLUTION 01 — Violation Identification & Safe Migration Steps
// Challenge 18: Normalise Schema Refactor
// ============================================================

// ---------------------------------------------------------
// PART 1: NORMALISATION VIOLATIONS
// ---------------------------------------------------------

// Violation 1: school_name, school_email, school_phone, school_region
// — Normal form broken: 2NF (partial dependency) and 3NF
//   (transitive dependency on a non-key attribute).
// — The school attributes do not depend on the booking's
//   primary key — they depend on the school entity itself.
// — Update anomaly: if a school changes its email, every
//   booking row for that school must be updated individually.
//   Missing one row = inconsistent data.
// — Insert anomaly: you cannot record a school's details
//   until it has made at least one booking.
// — Delete anomaly: deleting the last booking for a school
//   permanently destroys the school's contact information.

// Violation 2: trip_destination, trip_departure, trip_price
// — Same class of violation as school columns — trip data
//   belongs to a trip entity, not to a booking row.
// — Update anomaly: if a trip price changes, all bookings
//   for that trip show the wrong price unless every row
//   is updated. (Note: price snapshot is intentional —
//   see Solution 02 for the nuance.)
// — Delete anomaly: removing all bookings for a trip also
//   removes all record of the trip existing.

// Violation 3: payment_status, payment_amount, payment_ref
// — Concern mixing / Single Responsibility violation.
//   A booking can have multiple instalment payments; storing
//   one payment record inline means only one payment can
//   ever be recorded.
// — This is also a 1NF violation: a booking with two payments
//   would require repeating groups (payment_amount_1,
//   payment_amount_2 …) or overwriting the single column.
// — Solution: extract to a payments table with a FK back
//   to bookings.

// Violation 4: status as a free-text string with no constraint
// — No specific normal form, but a data-integrity violation.
// — Any string can be inserted: 'Paid', 'paid', 'PAID',
//   'paidd' are all different values to SQL but identical
//   in intent. Reporting queries break silently.
// — Solution: ENUM or a CHECK constraint to a known set
//   of values: pending, confirmed, cancelled, completed.

// Violation 5: No indexes at all
// — Not a normalisation issue but a serious performance
//   and query-correctness issue.
// — Every query (by school name, by status, by trip) does
//   a full table scan. At scale this is unacceptable.

// ---------------------------------------------------------
// PART 2: SAFE MIGRATION STEPS (ZERO-DOWNTIME APPROACH)
// ---------------------------------------------------------

// The goal is to never DROP a column until the new structure
// is fully populated and the application has been updated
// to use the new columns. This allows a rollback if needed.

// Migration File 1 — create new normalised tables
// — Create schools table: id, name, email, phone, region, timestamps.
// — Create trips table: id, destination, departure_date,
//   price_per_student (decimal 8,2), timestamps.
// — Create payments table: id, booking_id (FK), amount (decimal),
//   status (enum: pending|paid|refunded), reference (unique nullable),
//   paid_at (nullable timestamp), timestamps.
// — No FK columns added to bookings yet.

// Migration File 2 — backfill new tables from existing booking data
// — Use DB::table('bookings')->select(distinct school columns)->get()
//   to extract unique schools and insert into schools table.
// — Use same approach to extract unique trips by destination +
//   departure_date combination.
// — Extract payment data (where payment_amount IS NOT NULL)
//   and insert into payments with the correct booking_id.
// — This migration is data-only, no schema changes.

// Migration File 3 — add FK columns to bookings (nullable first)
// — Add school_id (unsignedBigInteger, nullable) to bookings.
// — Add trip_id (unsignedBigInteger, nullable) to bookings.
// — Add contact_email (string, nullable) to bookings.
//   This will hold the snapshotted school_email at time of booking.
// — Add booking_reference (string, unique nullable) to bookings.

// Migration File 4 — populate new FK columns
// — Loop through each booking row.
// — Find the matching school by school_name + school_email.
//   Set bookings.school_id = schools.id.
// — Find the matching trip by trip_destination + trip_departure.
//   Set bookings.trip_id = trips.id.
// — Copy school_email to contact_email (snapshot preservation).
// — Generate a booking_reference for rows that have none.
// — After all rows are updated, add NOT NULL constraints
//   and the FK constraint on school_id and trip_id.

// Migration File 5 — drop old denormalised columns (separate deploy)
// — Only run this migration after the application code has been
//   updated to read/write school_id and trip_id instead of
//   the inline school/trip columns.
// — Drop: school_name, school_email, school_phone, school_region.
// — Drop: trip_destination, trip_departure, trip_price.
// — Drop: payment_status, payment_amount, payment_ref.
// — Add CHECK constraint or change status column type to ENUM.
// — Add all missing indexes.

// ---------------------------------------------------------
// PART 3: INDEX ADDITIONS (MIGRATION FILE 5)
// ---------------------------------------------------------

// bookings: index on school_id, trip_id, status
// bookings: compound index on (trip_id, status) for capacity checks
// bookings: unique index on booking_reference
// payments: index on booking_id, status
// schools: index on region (used in region-based reporting)
// trips: index on departure_date, destination
